Simple helper function for padding numbers with a leading zero.

 
    pad: (i) ->
        (if i < 10 then "0"  else "") + i

    tick: ->
        @date = new Date( @epoch )

    tock: ()->
        t = @
        setInterval(
            ( ->
                t.epoch += t.refresh
                t.tick()
            ), @refresh 
        )

    getTime: ->
        [
            @pad( @date.getHours() ),
            @pad( @date.getMinutes() ),
            @pad( @date.getSeconds() )
        ].join( ':' )
